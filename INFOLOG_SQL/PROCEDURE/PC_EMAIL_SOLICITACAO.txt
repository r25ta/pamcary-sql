CREATE OR REPLACE
PROCEDURE PC_EMAIL_SOLICITACAO ( 
             PNOM_APLIC_SOL  VARCHAR2, 
             PDHR_SOLIC_ENV DATE, 
             PCOD_SITUA_ENV NUMBER, 
             PDHR_ENVIO DATE, 
             PDES_EMAIL_RES VARCHAR2, 
             PDES_EMAIL_ASS VARCHAR2, 
             PDES_EMAIL_TXT LONG, 
             PCOD_USER VARCHAR2, 
             PDHR_MANUT DATE ,
             PCTL_ENDER_SGP NUMBER )
AS
BEGIN
      INSERT INTO EMAIL_SOLICITACAO(CTL_SOLIC_ENV, NOM_APLIC_SOL, DHR_SOLIC_ENV, COD_SITUA_ENV, DHR_ENVIO,
      							    DES_EMAIL_RES, DES_EMAIL_ASS, DES_EMAIL_TXT, COD_USER, DHR_MANUT, CTL_ENDER_SGP)
      VALUES (1000, TO_MULTI_BYTE(PNOM_APLIC_SOL), TO_DATE(PDHR_SOLIC_ENV,'DD/MM/YYYY HH24:MI:SS'),
      		     PCOD_SITUA_ENV, TO_DATE(PDHR_ENVIO,'DD/MM/YYYY HH24:MI:SS'), TO_MULTI_BYTE(PDES_EMAIL_RES), TO_MULTI_BYTE(PDES_EMAIL_ASS),
      		     PDES_EMAIL_TXT, TO_MULTI_BYTE(PCOD_USER), TO_DATE(PDHR_MANUT,'DD/MM/YYYY HH24:MI:SS'), PCTL_ENDER_SGP );
END;
