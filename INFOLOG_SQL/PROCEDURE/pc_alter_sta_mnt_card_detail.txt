CREATE OR REPLACE PROCEDURE PC_ALTER_STA_MNT_CARD_DETAIL
AS
BEGIN
--
 UPDATE MONITORA_CARTAO_DETALHE SET
 sta_procs = 'I'
 WHERE ctl_plvia = 0
 AND sta_procs = 'N'
 AND TO_CHAR(dhr_alter,'YYYYMMDD') < TO_CHAR((SYSDATE - 1),'YYYYMMDD');
--
 COMMIT;
--
END PC_ALTER_STA_MNT_CARD_DETAIL;
