 CREATE OR REPLACE
 FUNCTION FC_EMAIL_COMUNICACAO_CONTATO( PCTL_VINCD NUMBER )
 
 RETURN NUMBER
 AS
 CURSOR
 EMAIL IS
   	
 cEMAIL EMAIL%ROWTYPE;  
 
 BEGIN
  OPEN EMAIL;
  FETCH EMAIL INTO cEMAIL;
  IF EMAIL%NOTFOUND THEN	 
 	 RETURN(NULL);
  END IF;
  CLOSE EMAIL;
  RETURN(cEMAIL.Qtd_real);
 END;
