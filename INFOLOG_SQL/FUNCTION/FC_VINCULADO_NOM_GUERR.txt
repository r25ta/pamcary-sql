CREATE OR REPLACE
FUNCTION FC_VINCULADO_NOM_GUERR (PCTL_VINCD NUMBER)
RETURN VARCHAR2
AS
CURSOR
 NOME IS
 SELECT nom_guerr
   FROM SUPERVISOR.VINCULADO
  WHERE ctl_vincd = PCTL_VINCD;
  RNOME NOME%ROWTYPE;
BEGIN
 OPEN NOME;
 FETCH NOME INTO RNOME;
 IF NOME%NOTFOUND THEN
    RETURN(NULL);
 END IF;
 CLOSE NOME;
 RETURN(RNOME.nom_guerr);
END;
/
